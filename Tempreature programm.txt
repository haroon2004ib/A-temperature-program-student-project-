#include <iostream>
using namespace std;

int main() {
    int temperature;

    cout << "Enter the temperature of water: ";
    cin >> temperature;

    if (temperature < 0) {
        cout << "Water is in solid state (ice)." << endl;
    }
    else if (temperature > 0 && temperature < 100) {
        cout << "Water is in liquid state." << endl;
    }
    else if (temperature > 100) {
        cout << "Water is in gaseous state (steam)." << endl;
    }
    else if (temperature == 0) {
        cout << "Water is at freezing/melting point." << endl;
    }
    else if (temperature == 100) {
        cout << "Water is at boiling point." << endl;
    }

    return 0;
}
